C Tricks Like JS

2022-03-21 · 1 分钟 · 25 字 · Author
#include <stdio.h>

int main()
{
    puts("-0.5" + 1);
    printf("%d\n", 50 ** "2");
    printf("%d\n", "(int)1e01" - "20");
    return 0;
}

/*
output:
0.5
2500
-10
*/
